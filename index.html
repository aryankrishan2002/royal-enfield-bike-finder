        // --- APPLICATION LOGIC ---

        document.addEventListener('DOMContentLoaded', () => {

            const body = document.querySelector('body');

            const header = document.getElementById('main-header');

            const mainContent = document.getElementById('app-container');

            const finderFlow = document.getElementById('finder-flow');

            

            const initialPage = document.getElementById('initial-page');

            const pages = finderFlow.querySelectorAll('[id^="page-"]');

            const startFinderBtn = document.getElementById('start-finder-btn');

            const motorcyclesNavLink = document.getElementById('motorcycles-nav-link');

            

            const showPage = (pageId) => {

                pages.forEach(page => page.classList.add('hidden'));

                const targetPage = document.getElementById(pageId);

                if(targetPage) {

                    targetPage.classList.remove('hidden');

                }

            };

            

            const resetToInitialPage = () => {

                finderFlow.style.display = 'none';

                mainContent.style.display = 'block';

                body.className = 'bg-white text-black';

                window.scrollTo(0, 0);

            };


            motorcyclesNavLink.addEventListener('click', (e) => {

                e.preventDefault();

                resetToInitialPage();

            });


            const populateInitialGrid = () => {

                const grid = document.getElementById('initial-bike-grid');

                if (!grid) return;

                grid.innerHTML = '';

                Object.values(bikes).forEach(bike => {

                    const bikeCard = document.createElement('a');

                    bikeCard.href = bike.link;

                    bikeCard.target = '_blank';

                    bikeCard.className = 'text-center group block';

                    bikeCard.innerHTML = `

                        <div class="overflow-hidden">

                           <img src="${bike.image}" alt="${bike.name}" class="w-full h-48 object-cover group-hover:scale-110 transition-transform duration-300">

                        </div>

                        <h3 class="font-semibold text-lg mt-2 text-gray-800">${bike.name}</h3>

                    `;

                    grid.appendChild(bikeCard);

                });

            };


            populateInitialGrid();


            startFinderBtn.addEventListener('click', () => {

                mainContent.style.display = 'none';

                finderFlow.style.display = 'block';

                body.className = 'bg-black text-white';

                showPage('page-start');

            });


            document.querySelectorAll('.finder-nav-btn').forEach(button => {

                button.addEventListener('click', () => {

                    const targetPageId = button.getAttribute('data-target');

                    showPage(targetPageId);

                });

            });


            document.querySelectorAll('.filter-btn').forEach(button => {

                button.addEventListener('click', () => {

                    const filterTag = button.getAttribute('data-filter');

                    renderResults(filterTag);

                    showPage('page-results');

                });

            });

            

            document.querySelectorAll('.start-again-btn').forEach(button => {

                button.addEventListener('click', () => {

                    showPage('page-start');

                });

            });


            const renderResults = (filterTag) => {

                const resultsGrid = document.getElementById('results-grid');

                resultsGrid.innerHTML = '';

                

                const filteredBikes = Object.values(bikes).filter(bike => bike.tags.includes(filterTag));

                

                filteredBikes.forEach((bike, index) => {

                    const card = document.createElement('div');

                    card.className = 'bg-gray-900 rounded-lg shadow-xl overflow-hidden flex flex-col card-enter';

                    card.style.animationDelay = `${index * 100}ms`;

                    

                    card.innerHTML = `

                        <img src="${bike.image}" alt="${bike.name}" class="w-full h-56 object-cover">

                        <div class="p-6 flex flex-col flex-grow">

                            <h3 class="text-2xl font-bold re-font mb-2">${bike.name}</h3>

                            <p class="text-gray-400 mb-4 text-sm flex-grow">${bike.note}</p>

                            <div class="space-y-2 text-sm mb-6 border-t border-b border-gray-700 py-4">

                                <p><strong>Engine:</strong> ${bike.engine}</p>

                                <p><strong>Power:</strong> ${bike.power}</p>

                                <p><strong>Weight:</strong> ${bike.weight}</p>

                                <p><strong>Seat Height:</strong> ${bike.seatHeight}</p>

                                <p><strong>Price (Ex-Showroom):</strong> ${bike.price}</p>

                            </div>

                            <div class="mt-auto flex gap-4">

                                <a href="${bike.link}" target="_blank" class="flex-1 text-center font-bold p-3 rounded-lg re-red re-red-hover transition-colors duration-300">Official Page</a>

                            </div>

                        </div>

                    `;

                    resultsGrid.appendChild(card);

                });

            };


            // Dropdown Logic

            const dropdowns = document.querySelectorAll('.dropdown');

            dropdowns.forEach(dropdown => {

                const toggle = dropdown.querySelector('.dropdown-toggle');

                toggle.addEventListener('click', (event) => {

                    event.stopPropagation();

                    // Close other dropdowns

                    dropdowns.forEach(d => {

                        if (d !== dropdown) {

                            d.classList.remove('open');

                        }

                    });

                    // Toggle current dropdown

                    dropdown.classList.toggle('open');

                });

            });


            // Close dropdowns when clicking outside

            window.addEventListener('click', () => {

                dropdowns.forEach(dropdown => {

                    dropdown.classList.remove('open');

                });

            });

        });
